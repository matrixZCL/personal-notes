```js
    <script>
      //1.利用函数关键字声明函数
      function fn(){

      }

      //2.函数表达式（匿名函数）
      //var 变量名 = function(){};
      var hello = function(hello){
          console.log(hello);
          
      }
      hello('hello world');
      //hello是变量名，不是函数名
      //函数表达式声明方式跟变量差不多，变量存的是值，函数表达式存的是函数
      //可以参数传递
    </script>
```
